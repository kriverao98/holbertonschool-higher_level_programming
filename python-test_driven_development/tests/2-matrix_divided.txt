import unittest
from 2-matrix_divided import matrix_divided

class TestMatrixDivided(unittest.TestCase):
    def test_matrix_divided(self):
        # Test with valid input
        matrix = [[1, 2, 3], [4, 5, 6]]
        result = matrix_divided(matrix, 2)
        self.assertEqual(result, [[0.5, 1.0, 1.5], [2.0, 2.5, 3.0]])

        # Test with invalid matrix
        with self.assertRaises(TypeError):
            matrix_divided(123, 2)
        with self.assertRaises(TypeError):
            matrix_divided([1, 2, 3], 2)

        # Test with invalid div
        with self.assertRaises(TypeError):
            matrix_divided(matrix, "2")
        with self.assertRaises(ZeroDivisionError):
            matrix_divided(matrix, 0)

        # Test with uneven matrix
        uneven_matrix = [[1, 2, 3], [4, 5]]
        with self.assertRaises(TypeError):
            matrix_divided(uneven_matrix, 2)

if __name__ == '__main__':
    unittest.main()
